        return name;
    }

    public void setName(String name) {
        if (name.length() >= 10) {
            this.name = name;
        } else {

        }

    public String getId() {
        return id;
    }

    public void setId(String id) {

        } else {

    public String getPhone() {

    }

    public void setPhone(String phone) {

        } else {
            throw new IllegalArgumentException("Invalid phone");

        return age;
    }

    public void setAge(int age) {
